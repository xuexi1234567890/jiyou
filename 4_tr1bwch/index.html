LED指令：

import machine
import time

# 初始化I2C
i2c = machine.I2C(1, scl=machine.Pin(15), sda=machine.Pin(21))

# I2C设备地址和寄存器地址
i2c_addr = 0x40
reg_addr = 160  # 0xA0

# 往寄存器写入0和1，交替间隔1秒
value = 0
while True:
    i2c.writeto_mem(i2c_addr, reg_addr, bytes([value]))
    value = 1 - value  # 交替0和1
    time.sleep(1)
